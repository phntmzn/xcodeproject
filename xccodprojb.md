def build_pbxproj(app_name: str, bundle_id: str) -> str:
    return (
        PBXPROJ_TEMPLATE
        .replace("__APP_NAME__", app_name)
        .replace("__BUNDLE_ID__", bundle_id)
    )


def reset_output() -> None:
    if ROOT.exists():
        shutil.rmtree(ROOT)
    APP_DIR.mkdir(parents=True, exist_ok=True)
    SHARED_DIR.mkdir(parents=True, exist_ok=True)
    RES_DIR.mkdir(parents=True, exist_ok=True)
    WORKSPACE_DIR.mkdir(parents=True, exist_ok=True)


def write_files() -> None:
    (APP_DIR / f"{APP_NAME}App.swift").write_text(APP_SWIFT, encoding="utf-8")
    (SHARED_DIR / "ContentView.swift").write_text(CONTENT_VIEW_SWIFT, encoding="utf-8")
    (OUT_README := ROOT / "README.md").write_text(README_MD, encoding="utf-8")
    (PROJ_DIR / "project.pbxproj").write_text(build_pbxproj(APP_NAME, BUNDLE_ID), encoding="utf-8")
    (WORKSPACE_DIR / "contents.xcworkspacedata").write_text(WORKSPACE_XML, encoding="utf-8")


def verify() -> None:
    result = subprocess.run(
        ["xcodebuild", "-list", "-project", str(PROJ_DIR)],
        capture_output=True,
        text=True,
    )
    print(result.stdout.strip())
    if result.stderr.strip():
        print(result.stderr.strip())
    if result.returncode != 0:
        raise SystemExit(result.returncode)


def main() -> None:
    reset_output()
    write_files()

    print(f"Created project at: {ROOT}")
    print(f"Open in Xcode: {PROJ_DIR}")
    print("Verifying with xcodebuild...\n")
    verify()


if __name__ == "__main__":
    main()

Change these at the top:

```python
APP_NAME = "NewMacApp"
BUNDLE_ID = "com.example.NewMacApp"
ROOT = Path.home() / "Desktop" / APP_NAME
```

Run:

```bash
python3 new_xcode_project.py
```

This makes a brand new macOS Xcode project bundle with:
- `App/`
- `Shared/`
- `Resources/`
- `NewMacApp.xcodeproj/`
- `project.xcworkspace/contents.xcworkspacedata`

Then verifies it with `xcodebuild -list`.
